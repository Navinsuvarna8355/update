/* ══════════════════════════════════════════════════════ NAVINS TERMINAL - HYPER SENSITIVE ENGINE (V3) Fixes: Sideways Trap on Expiry Afternoon ══════════════════════════════════════════════════════ */ function updateOptionsBattle() { if (!chainData || !chainData.length) return; const spot = chainData[0].underlying_spot_price; const instrKey = document.getElementById('symbolSel').value; const step = INSTRUMENTS[instrKey]?.step || 50; const atm = Math.round(spot / step) * step; // Time Sensitivity: 1:30 PM ke baad logic badal jayega const now = new Date(); const isAfternoon = (now.getHours() > 13 || (now.getHours() === 13 && now.getMinutes() >= 30)); let ceWriters = 0, ceBuyers = 0, peWriters = 0, peBuyers = 0; let maxCOI = 0, maxPOI = 0, resStrike = 0, supStrike = 0; let atmCE = null, atmPE = null; chainData.forEach(d => { const ce = d.call_options || {}; const pe = d.put_options || {}; const s = d.strike_price; if (Math.abs(s - spot) <= 750) { if ((ce.market_data?.oi || 0) > maxCOI) { maxCOI = ce.market_data.oi; resStrike = s; } if ((pe.market_data?.oi || 0) > maxPOI) { maxPOI = pe.market_data.oi; supStrike = s; } } if (s === atm) { atmCE = ce; atmPE = pe; } // BATTLE FORCE CALCULATION const ceChg = (ce.market_data?.oi || 0) - (ce.market_data?.prev_oi || 0); const peChg = (pe.market_data?.oi || 0) - (pe.market_data?.prev_oi || 0); const ceLtpChg = ((ce.market_data?.ltp - (ce.market_data?.close_price || ce.market_data?.ltp)) / (ce.market_data?.close_price || 1)) * 100; const peLtpChg = ((pe.market_data?.ltp - (pe.market_data?.close_price || pe.market_data?.ltp)) / (pe.market_data?.close_price || 1)) * 100; if (Math.abs(s - spot) <= 500) { // Afternoon mein "OI Change" aur "LTP Speed" ka weight 2 guna badha diya const weight = isAfternoon ? 2.5 : 1.0; if (ceLtpChg < -0.01) { ceWriters += (Math.abs(ceChg) * weight); } else { ceBuyers += (Math.abs(ceChg) * weight); } if (peLtpChg < -0.01) { peWriters += (Math.abs(peChg) * weight); } else { peBuyers += (Math.abs(peChg) * weight); } } }); // 🔮 HYPER VERDICT ENGINE const vEl = document.getElementById('marketVerdict'); const cwEl = document.getElementById('callWinner'); const pwEl = document.getElementById('putWinner'); // Winner Labels Update cwEl.innerHTML = ceWriters > ceBuyers ? 'WRITERS DOMINANT 🔴' : 'BUYERS DOMINANT 🟢'; pwEl.innerHTML = peWriters > peBuyers ? 'WRITERS DOMINANT 🟢' : 'BUYERS DOMINANT 🔴'; // Proportional Battle Ratios const ceForce = ceWriters / (ceBuyers || 1); const peForce = peBuyers / (peWriters || 1); if (ceForce > 1.2 && peForce > 1.2) { vEl.textContent = "STRONG BEARISH 📉"; vEl.style.color = "#ff3d6b"; } else if (peWriters > peBuyers * 1.2 && ceBuyers > ceWriters * 1.2) { vEl.textContent = "STRONG BULLISH 🚀"; vEl.style.color = "#00e5a0"; } else if (isAfternoon && ceWriters > ceBuyers * 1.1) { vEl.textContent = "BEARISH PRESSURE ↘️"; vEl.style.color = "#ff9f40"; } else { vEl.textContent = "SIDEWAYS / RANGE ⚖️"; vEl.style.color = "#ffd000"; } // Numbers display const fNum = (n) => n >= 100000 ? (n/100000).toFixed(2) + 'L' : (n/1000).toFixed(1) + 'K'; document.getElementById('ceWriteVal').textContent = fNum(ceWriters); document.getElementById('ceBuyVal').textContent = fNum(ceBuyers); document.getElementById('peWriteVal').textContent = fNum(peWriters); document.getElementById('peBuyVal').textContent = fNum(peBuyers); document.getElementById('strongResist').textContent = resStrike; document.getElementById('strongSupport').textContent = supStrike; }
